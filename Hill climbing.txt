graph = {
    'A': [('B', 3), ('C', 5)],
    'B': [],
    'C': [('D', 8)],
    'D': []
}

def hill_climb(start, goal):
    current = start
    path = [current]
    while current != goal:
        neighbors = graph[current]
        if not neighbors:
            break
        next_node = max(neighbors, key=lambda x: x[1])[0]
        path.append(next_node)
        current = next_node
    print("Path:", " â†’ ".join(path))

hill_climb('A', 'D')
